\newglossaryentry{energy}{
	name={Energy},
	text={Energy},
	description={The capacity of a system to perform work or produce change, typically measured in joules (J). Energy exists in various forms, including kinetic, potential, thermal, and chemical energy, and can be transferred or transformed but not created or destroyed (according to the law of conservation of energy)}
}
\newglossaryentry{joule}{
	name={Joule},
	text={joule},
	description={The SI unit of energy, represented by the symbol \( J \). One joule is defined as the amount of work done when a force of one newton displaces an object by one meter in the direction of the force, equivalent to \( 1 \, \text{J} = 1 \, \text{N} \cdot \text{m} \)}
}
\newglossaryentry{mechanicalenergy}{
	name={Mechanical Energy},
	text={mechanical energy},
	description={The sum of kinetic and potential energy in a system, representing the energy associated with the motion and position of an object. Mechanical energy is conserved in an isolated system with no non-conservative forces, such as friction}
}
\newglossaryentry{kineticenergy}{
	name={Kinetic Energy},
	text={Kinetic energy},
	description={The energy an object possesses due to its motion, calculated as \( KE = \frac{1}{2}mv^2 \), where \( m \) is the object's mass and \( v \) is its velocity. Kinetic energy depends on both mass and speed, and it increases with the square of the speed}
}

\newglossaryentry{gravitationalpotentialenergy}{
	name={Gravitational Potential Energy},
	text={Gravitational potential energy},
	description={The energy an object possesses due to its position in a gravitational field, commonly calculated as \( PE = mgh \), where \( m \) is the object's mass, \( g \) is the acceleration due to gravity, and \( h \) is the height above a reference point}
}

\newglossaryentry{elasticpotentialenergy}{
	name={Elastic Potential Energy},
	text={elastic potential energy},
	description={The energy stored in an elastic material, such as a spring, when it is stretched or compressed. Elastic potential energy is calculated as \( PE = \frac{1}{2}kx^2 \), where \( k \) is the spring constant and \( x \) is the displacement from the equilibrium position}
}

\newglossaryentry{pressure}{
	name={Pressure},
	text={pressure},
	description={A physical quantity defined as the force exerted per unit area. It is mathematically expressed as \( P = \frac{F}{A} \), where \( P \) is pressure, \( F \) is the applied force, and \( A \) is the area over which the force is distributed. The SI unit of pressure is the pascal (Pa), where \( 1 \, \text{Pa} = 1 \, \text{N/m}^2 \)}
}
\newglossaryentry{hydrostaticpressure}{
	name={Hydrostatic Pressure},
	text={Hydrostatic pressure},
	description={The pressure exerted by a fluid at rest due to the force of gravity. It is given by the equation \( P = \rho g h \), where \( P \) is the hydrostatic pressure, \( \rho \) is the fluid density, \( g \) is the acceleration due to gravity, and \( h \) is the depth of the fluid. Hydrostatic pressure increases with depth and acts equally in all directions at a given point in the fluid}
}
\newglossaryentry{gaugepressure}{
	name={Pressure, Gauge},
	text={gauge pressure},
	description={The pressure measured relative to atmospheric pressure. It is defined as \( P_{\text{gauge}} = P_{\text{absolute}} - P_{\text{atm}} \), where \( P_{\text{absolute}} \) is the absolute pressure and \( P_{\text{atm}} \) is the atmospheric pressure. Gauge pressure is positive when it is above atmospheric pressure and negative when it is below (such as in a vacuum)}
}

\newglossaryentry{absolutepressure}{
	name={Pressure, Absolute},
	text={absolute pressure},
	description={The total pressure measured relative to a perfect vacuum (zero pressure). It is given by \( P_{\text{absolute}} = P_{\text{gauge}} + P_{\text{atm}} \), where \( P_{\text{gauge}} \) is the gauge pressure and \( P_{\text{atm}} \) is the atmospheric pressure. Absolute pressure is always positive and is used in many scientific and engineering calculations}
}
\newglossaryentry{density}{
	name={Density},
	text={density},
	description={A measure of mass per unit volume of a substance. It is mathematically expressed as \( \rho = \frac{m}{V} \), where \( \rho \) (rho) is the density, \( m \) is the mass, and \( V \) is the volume. The SI unit of density is kilograms per cubic meter (\(\text{kg/m}^3\)), though grams per cubic centimeter (\(\text{g/cm}^3\)) is also commonly used}
}


\newglossaryentry{potentialenergy}{
	name={Potential Energy},
	text={Potential energy},
	description={The stored energy in a system due to the position, arrangement, or state of its components. Potential energy can take various forms, including gravitational potential energy (due to height in a gravitational field) and elastic potential energy (due to stretching or compressing an elastic material)}
}
\newglossaryentry{hookeslaw}{
	name={Hooke's Law},
	text={Hooke's law},
	description={A principle describing the behavior of springs and other elastic materials, stating that the force exerted by a spring is proportional to its displacement from the equilibrium position. Mathematically, \( F = -kx \), where \( F \) is the restoring force, \( k \) is the spring constant, and \( x \) is the displacement}
}
\newglossaryentry{springconstant}{
	name={Spring Constant},
	text={spring constant},
	description={A parameter that quantifies the stiffness of a spring or elastic material. It is denoted by \( k \) and has units of force per unit length (e.g., \(\text{N/m}\)). The spring constant determines the proportionality between the restoring force exerted by the spring and its displacement from the equilibrium position, as described by Hooke's Law}
}

\newglossaryentry{work}{
	name={Work},
	text={Work},
	description={A measure of energy transfer that occurs when an object is moved by a force. Calculated as \( W = Fd \cos \theta \), where \( W \) is the work done, \( F \) is the applied force, \( d \) is the displacement, and \( \theta \) is the angle between the force and displacement directions}
}

\newglossaryentry{latentheatfusion}{
	name={Heat, Latent Heat of Fusion},
	text={latent heat of fusion},
	description={The amount of heat required to change a substance from a solid to a liquid at constant temperature and pressure. It is denoted by \( L_f \) and measured in joules per kilogram (\(\text{J/kg}\)). The latent heat of fusion represents the energy needed to overcome intermolecular forces during melting without changing the temperature}
}

\newglossaryentry{latentheatvaporization}{
	name={Heat, Latent Heat of Vaporization},
	text={latent heat of vaporization},
	description={The amount of heat required to change a substance from a liquid to a gas at constant temperature and pressure. It is denoted by \( L_v \) and measured in joules per kilogram (\(\text{J/kg}\)). The latent heat of vaporization represents the energy needed to break intermolecular bonds completely during vaporization without changing the temperature}
}

\newglossaryentry{wave}{
	name={Wave},
	text={wave},
	description={A disturbance that transfers energy through space or a medium without a net movement of matter. Waves can be classified into mechanical and electromagnetic waves, depending on whether they require a medium for propagation}
}

\newglossaryentry{medium}{
	name={Medium},
	text={medium},
	description={The substance or material through which a mechanical wave travels. Examples include air for sound waves, water for ocean waves, and solid materials for seismic waves}
}

\newglossaryentry{transversewave}{
	name={Wave, Transverse},
	text={Transverse Wave},
	description={A type of wave in which the disturbance is perpendicular to the direction of wave propagation. Examples include water waves and electromagnetic waves. In a transverse wave, crests and troughs represent areas of maximum displacement}
}

\newglossaryentry{longitudinalwave}{
	name={Wave, Longitudinal},
	text={Longitudinal Wave},
	description={A type of wave in which the disturbance is parallel to the direction of wave propagation. Examples include sound waves and pressure waves in fluids. Longitudinal waves consist of compressions and rarefactions}
}

\newglossaryentry{electromagneticwave}{
	name={Wave, Electromagnetic},
	text={Electromagnetic Wave},
	description={A wave that consists of oscillating electric and magnetic fields and does not require a medium to propagate. Electromagnetic waves travel at the speed of light in a vacuum and include radio waves, microwaves, infrared, visible light, ultraviolet, X-rays, and gamma rays}
}
\newglossaryentry{amplitude}{
	name={Amplitude},
	text={Amplitude},
	description={The maximum displacement of a wave from its equilibrium position. It represents the wave's energy and is measured in units appropriate to the wave type (e.g., meters for mechanical waves, volts per meter for electromagnetic waves)}
}

\newglossaryentry{period}{
	name={Period},
	text={period},
	description={The time required for one complete cycle of a wave to pass a given point. It is denoted by \( T \) and is measured in seconds (s). The period is related to frequency by the equation \( T = \frac{1}{f} \)}
}

\newglossaryentry{frequency}{
	name={Frequency},
	text={frequency},
	description={The number of complete wave cycles that pass a given point per unit time. It is denoted by \( f \) and is measured in hertz (Hz), where \( 1 \text{ Hz} = 1/\text{s} \). Frequency and period are related by \( f = \frac{1}{T} \)}
}

\newglossaryentry{wavelength}{
	name={Wavelength},
	text={wavelength},
	description={The distance between two consecutive points in phase on a wave, such as crest to crest or trough to trough in a transverse wave. It is denoted by \( \lambda \) (lambda) and is measured in meters (m). The wavelength is related to wave speed and frequency by \( v = f\lambda \)}
}

\newglossaryentry{resonance}{
	name={Resonance},
	text={Resonance},
	description={A phenomenon that occurs when an external force matches the natural frequency of a system, causing it to oscillate with maximum amplitude. Resonance is observed in mechanical systems, electrical circuits, and acoustic waves, and is responsible for phenomena such as the amplification of sound in musical instruments}
}

\newglossaryentry{node}{
	name={Node},
	text={node},
	description={A point in a standing wave where there is no displacement due to destructive interference. Nodes remain stationary and occur at fixed positions along the medium in systems with standing waves, such as vibrating strings and air columns}
}

\newglossaryentry{antinode}{
	name={Antinode},
	text={antinode},
	description={A point in a standing wave where the displacement is maximum due to constructive interference. Antinodes alternate with nodes along the medium and represent points of highest oscillation in systems such as vibrating strings and sound waves}
}

\newglossaryentry{standingwave}{
	name={Standing Wave},
	text={standing wave},
	description={A wave pattern that results from the interference of two waves traveling in opposite directions, creating stationary nodes and oscillating antinodes. Standing waves form in bounded systems such as strings, air columns, and electromagnetic cavities, and are fundamental to the study of wave resonance}
}

\newglossaryentry{harmonic}{
	name={Harmonic},
	text={harmonic},
	description={A frequency component of a wave that is an integer multiple of the fundamental frequency. Harmonics are observed in vibrating systems such as strings, air columns, and electrical circuits. The first harmonic (fundamental frequency) corresponds to the lowest frequency standing wave, while higher harmonics (overtones) correspond to additional modes of vibration}
}

\newglossaryentry{heat}{
	name={Heat},
	text={Heat},
	description={A form of energy transfer due to a temperature difference between two objects or systems. Heat flows spontaneously from higher temperature to lower temperature regions and is measured in joules (\si{J}) in the SI system}
}

\newglossaryentry{conduction}{
	name={Conduction},
	text={Conduction},
	description={The transfer of heat through a material without the bulk movement of matter. It occurs due to collisions between particles in a solid or fluid and is governed by Fourier's law, where the heat transfer rate is proportional to the temperature gradient}
}

\newglossaryentry{convection}{
	name={Convection},
	text={Convection},
	description={The transfer of heat through the bulk movement of a fluid (liquid or gas). Convection can be natural (due to density differences) or forced (due to external forces such as fans or pumps). It plays a key role in atmospheric circulation, ocean currents, and heat dissipation in engineering systems}
}

\newglossaryentry{thermalradiation}{
	name={Thermal Radiation},
	text={Thermal radiation},
	description={The transfer of heat in the form of electromagnetic waves, primarily infrared radiation. Unlike conduction and convection, thermal radiation does not require a medium for transfer. It follows Stefan-Boltzmannâ€™s law, which states that the power radiated by an object is proportional to the fourth power of its absolute temperature}
}

\newglossaryentry{adiabatic}{
	name={Adiabatic Process},
	text={adiabatic process},
	description={A thermodynamic process in which no heat is exchanged with the surroundings (\( Q = 0 \)). Changes in internal energy result only from work done on or by the system. Adiabatic processes are common in rapid expansions or compressions, such as in gas dynamics and atmospheric science}
}

\newglossaryentry{isothermal}{
	name={Isothermal Process},
	text={isothermal process},
	description={A thermodynamic process in which the temperature of the system remains constant (\( \Delta T = 0 \)). Any heat added to the system is fully converted into work, maintaining a constant internal energy. Isothermal processes often occur in systems that exchange heat with a large thermal reservoir}
}

\newglossaryentry{isochoric}{
	name={Isochoric Process},
	text={isochoric process},
	description={A thermodynamic process in which the volume remains constant (\( \Delta V = 0 \)), meaning no work is done by the system. Any heat transfer results in a change in internal energy. Isochoric processes are common in rigid, sealed containers where expansion is restricted}
}

\newglossaryentry{isobaric}{
	name={Isobaric Process},
	text={isobaric process},
	description={A thermodynamic process in which the pressure remains constant (\( \Delta P = 0 \)). Any heat added to the system results in a change in both internal energy and volume. Isobaric processes are common in atmospheric and engineering applications, such as heating gases at constant pressure}
}

\newglossaryentry{halflife}{
	name={Half-Life},
	text={Half-life},
	description={The time required for half of a radioactive substance to decay. It is a characteristic property of a radioactive isotope and is denoted by \( t_{1/2} \). The remaining quantity of a radioactive substance after \( n \) half-lives is given by \( N = N_0 \left( \frac{1}{2} \right)^n \)}
}

\newglossaryentry{alphadecay}{
	name={Alpha Decay},
	text={Alpha decay},
	description={A type of radioactive decay in which an unstable nucleus emits an alpha particle (\( ^4_2 \text{He} \)), reducing its atomic number by 2 and mass number by 4. Alpha decay occurs in heavy elements such as uranium and radium}
}

\newglossaryentry{betadecay}{
	name={Beta Decay},
	text={beta decay},
	description={A type of radioactive decay in which a nucleus emits a beta particle, which can be either an electron (\( \beta^- \)) or a positron (\( \beta^+ \)). Beta decay occurs when a neutron converts into a proton (\( \beta^- \) decay) or a proton converts into a neutron (\( \beta^+ \) decay), changing the atomic number of the nucleus}
}

\newglossaryentry{gammadecay}{
	name={Gamma Decay},
	text={gamma decay},
	description={A type of radioactive decay in which an excited nucleus releases excess energy by emitting a gamma photon (\( \gamma \)). Gamma decay does not change the atomic number or mass number of the nucleus but reduces its energy state}
}

\newglossaryentry{alphaparticle}{
	name={Alpha Particle},
	text={alpha particle},
	description={A positively charged particle consisting of two protons and two neutrons, identical to a helium-4 nucleus (\( ^4_2 \text{He} \)). Alpha particles are emitted in alpha decay and have low penetration power, being stopped by paper or human skin}
}

\newglossaryentry{betaparticle}{
	name={Beta Particle},
	text={beta particle},
	description={A high-energy electron (\( \beta^- \)) or positron (\( \beta^+ \)) emitted in beta decay. Beta particles have greater penetration power than alpha particles and can be stopped by a few millimeters of aluminum}
}

\newglossaryentry{gammaparticle}{
	name={Gamma Particle},
	text={gamma particle},
	description={A high-energy photon (\( \gamma \)) emitted in gamma decay. Gamma radiation has no charge or mass and has very high penetration power, requiring dense materials like lead for shielding}
}

\newglossaryentry{element}{
	name={Element},
	text={element},
	description={A pure substance consisting of only one type of atom, characterized by a unique number of protons in its nucleus (atomic number, \( Z \)). Elements are the fundamental building blocks of matter and are organized in the periodic table}
}

\newglossaryentry{isotope}{
	name={Isotope},
	text={isotope},
	description={A variant of an element that has the same number of protons but a different number of neutrons. Isotopes of an element share the same chemical properties but may have different atomic masses and nuclear stability. Some isotopes are stable, while others are radioactive}
}

\newglossaryentry{ion}{
	name={Ion},
	text={ion},
	description={An atom or molecule that has gained or lost one or more electrons, resulting in a net electric charge. A positively charged ion is called a cation, while a negatively charged ion is called an anion. Ions play essential roles in chemical reactions, conductivity, and biological processes}
}
