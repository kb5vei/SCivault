	\item You step off the 5-m high-dive into a pool.
	\begin{enumerate}
		\vspace{-0.1in} \item How long does it take you to hit the water?
		
			\color{red}
			\begin{center}
		 \hspace{-0.5in} $\downarrow+ \hspace{0.5in} d = \cancelto{0}{v_it} + \frac{1}{2}at^2 \Longrightarrow t = \sqrt{\frac{2d}{a}} = \sqrt{\frac{2 (5m)} {9.81 m/s^2}} \approx 1.010 s
			$
			\end{center}
			\color{black}
		
		
		\item What is the speed that you hit the water at?

				\color{red}
			\begin{center}
			 \hspace{-0.5in} $\downarrow+ \hspace{0.5in}  v_f = v_i + at = 0 m/s + (9.81 m/s) (1.010s) \approx 9.905 m/s
				$
			\end{center}
			\color{black}

	\end{enumerate}
		\vspace{-.1in}	\item Ashley drops a rock off of a bridge.  It takes the rock 4.81 seconds to hit the ground.  How tall is the bridge?  
	
		\color{red}
		\begin{center}
			\vspace{-.1in}
		 \hspace{-0.5in} $\downarrow+ \hspace{0.5in}  d = \cancelto{0}{v_it} + \frac{1}{2}at^2  = \frac{1}{2}(9.81m/s^2)(4.81s)^2 \approx 114.483m $
		\end{center}
		\color{black}
	
	
	
	\item A monkey is on the ground.  He jumps at a velocity of 10 m/s directly up.  
		\begin{enumerate}
		\vspace{-.1in} \item What is the time that it takes the monkey to reach the top his jump?
			
			\color{red}
			\begin{center}
				\hspace{-1in} $\uparrow+ \hspace{0.5in}  v_f = v_i + a t \Longrightarrow t = \frac{v_f-v_i}{a} = \frac{0m/s - 10m/s}{-9.81m/s^2} \approx 1.019 s 
				$
			\end{center}
			\color{black}
			
		
		
		
		\item How high does he jump?

				
			\color{red}
			\begin{center}
				\hspace{-1in} $\uparrow+ \hspace{0.5in}  d = v_i t + \frac{1}{2}at^2 = 10 m/s \cdot 1.019s+ \frac{1}{2}(-9.81m/s^2)(1.019s)^2 \approx 5.097m
				$
			\end{center}
			\color{black}

		\item What is the total time that the monkey was in the air?
					
	\color{red}
	\begin{center}
	$t_{total} = 2 \cdot t_{up}  = 	2 \cdot 1.019s \approx 2.039 s	$
	\end{center}
	\color{black}
	




	\end{enumerate}
