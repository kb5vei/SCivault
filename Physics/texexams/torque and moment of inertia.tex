	\choice {Steering the right decreases the friction on your car, making it slide better.}

		\choice {A force that causes an object to rotate.}

	\end{question}

	\begin{question}
	 	Torque can be calculated using which formulas? (CHOOSE TWO)
		\choice {$ \omega_f = \vec {\tau} \times \vec{\theta}$}
		\choice [!]{$\tau = I \alpha$}
		\choice {$ F = m a$}
		\choice [!]{$\tau = r \cdot F \cdot cos(\theta)$ }
	\end{question}


\begin{question}
	Which of the following would cause the torque acting on an object to be $0 m\times N$?? (CHOOSE TWO)
	\choice {The force is applied to anywhere other than the center of mass.}
	\choice [!]{The force is applied to the center of mass.}
	\choice [!]{The force is applied in a direction parallel to the radius.}
	\choice {The object has a non-zero angular acceleration.}
\end{question}

\begin{question}

	\choice {its moment of inertia decreases.}
	\choice [!]{its moment of inertia increases.}
	\choice [!]{any torques on the balloon increase.}
	\choice {any torques on the balloon decrease.}
\end{question}

\begin{question}

	\choice [!]{An object rotating at a constant velocity ($\omega)$.}

\end{question}
