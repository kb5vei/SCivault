\begin{multiplechoice} [title={Multiple Choice},
	rearrange=no]
	\textit{Choose the best answer to each question.} 
	
	\begin{question}
	Which of the following numbers is in proper scientific notation?
	\choice{$82 \times 10^7$}	
	\choice[!]{$8.2 \times 10^8$}
	\choice{$8.2$E$8$}
	\choice{$0.82 \times 10^9$}
	\end{question}
	
	\begin{question}
	Which of the following is 42579624489 in scientific notation?
	\choice{$4.2579624489 \times 10^{10}$}	
	\choice[!]{$ 4.258 \times 10^{10}$}
	\choice{$ 42.580 \times 10^{9}$}
	\choice{$ 0.425 \times 10^{-11}$}	
	\end{question}
		
	\begin{question}
	What is $ (2 \times 10^3) (6 \times 10^{-7}) $ in proper scientific notation?
	\choice [!]{$1.2 \times 10^{-3} $ }
	\choice {$12 \times 10^{-4}$}
	\choice {$8 \times 10^{21}$}
	\choice {$4 \times 10^{-10}$ }
\end{question}


	\begin{question}
	What is $ \frac{9 \times 10^3}{ -3 \times 10^{8}} $ in proper scientific notation?
	\choice {$3 \times 10^5 $ }
	\choice {$-3 \times 10^5 $}
	\choice {$3 \times 10^{-5} $}
	\choice [!] {$-3 \times 10^{-5} $}
\end{question}

\begin{question}
	Simplify the following expression.  Express your answer in proper scientific notation.
	\begin{equation*}
		\frac{(6 \times 10^{-37})^2}{(2 \times 10^{10})^2} 
	\end{equation*}
	\choice {$3 \times 10^{-47}$ }
	\choice {$1.8 \times 10^{-83} $}
	\choice [!]{$9 \times 10^{-94}$}
	\choice {$8.88 \times 10^{23}$}
\end{question}

\begin{question}
	Simplify the following expression.  Express your answer in proper scientific notation.
	\begin{equation*}
		\frac{(4.357 \times 10^{17})^3}{(5.452 \times 10^{-7})\times (2.997 \times 10^8)} 
	\end{equation*}
	\choice {$2.667 \times 10^{15} $}
	\choice [!]{$5.062 \times 10^{50}$ }
	\choice {$4.547 \times 10^{67}$}
	\choice {None of the above}
\end{question}

\begin{question}
	The distance from El Paso to Dallas is 917 kilometers.  What is this distance in meters?
	\choice[!] {$9.17 \times 10^{5}$ m}
	\choice {$917 \times 10^{3}$ m }
	\choice {$9.17 \times 10^{3}$ m}
	\choice {$7.19 \times 10^{-5}$ m}
\end{question}



\end{multiplechoice}
