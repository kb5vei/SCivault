	 \choice {Ozone Layer}
	 \choice {Nitrogen}
	 \choice [!]{Greenhouse Gasses}
	\end{question}

	\begin{question}
	Which gas is most responsible for trapping heat in the atmosphere?
	\choice {Oxygen}
	\choice {Nitrogen}
	\choice [!]{Carbon Dioxide}
	\choice {Ozone}
\end{question}

	\begin{question}
How do we know that carbon dioxide levels are rising? 
	\choice [!]{We have directly measured the level of carbon dioxide in the air for 60 years.}
	\choice {We know that animals breathe out Carbon Dioxide.}
	\choice {We know that we are cutting down the rainforests.  Trees turn carbon dioxide into oxygen, so less trees means more carbon dioxide.}
	\choice {It is a scientific hypothesis that has not been proven.}
\end{question}

\begin{question}
	How have scientists been able to gather data about the carbon dioxide in the atmosphere for the last 800,000 years?
	\choice{They drilled into the bottom of the ocean to look for shells.}
	\choice[!]{They drilled into the ice in Antarctica to look for bubbles of trapped air.}
	\choice{They ran computer simulations about the past.}
	\choice{They took the observations of the last 60 years and extended the trend backward.}
\end{question}

\begin{question}
	Which part of the climate system has been responsible for absorbing most of the extra heat?
		\choice{The Land} 
		\choice[!]{The Oceans}
		\choice{The Atmosphere}
		\choice{The Ice in the polar caps.}
\end{question}

\begin{question}
How do we know that burning fossil fuels is the cause for the extra carbon dioxide in our atmosphere? 
	\choice[!]{Carbon comes in different forms.  By determining what forms of carbon are in the air, we can determine their source. }
	\choice{We have done the math and calculated how much carbon dioxide should be formed by burning fossil fuels.}
	\choice{We don't know if the carbon dioxide in the air is caused by humans, or other natural causes such as volcanoes.}
	\choice{Carbon dioxide in the atmosphere increases in the winter and decreases in the summer because of trees.}

\end{question}

\begin{question}
How do scientists determine the temperatures of the earth a very long time ago?
		\choice [!]{They look at shells from sea creatures such as clams.  The chemical composition of the shell is changed by the temperature of the water. }
		\choice {They determine how close the earth is to the sun through computer simulations.}
		\choice {They dig up dinosaur bones and test the chemicals they are made of.}
		\choice {There is no way to tell what the temperature of the earth was a very long time ago.}
\end{question}

\begin{question}
	Historically, how have carbon dioxide concentration and temperatures been related? 
		\choice [!]{They are very closely correlated.  When one goes up the other does as well.}
		\choice {They are inversely related. When one goes up the other goes down. }
		\choice {There is no relationship between the two.}
		\choice{We have no data about this relationship.}
\end{question}
