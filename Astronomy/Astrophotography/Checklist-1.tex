	\normalsize

	
	
	\begin{itemize}
		\item Columnation
		\item Physical Setup
		
		\begin{todolist}
			\item Tripod
			\item Mount
			\begin{todolist}
				\item Roughly pointed North.
				\item Leveled
			\end{todolist}
			\item Counterweights and endcap
			\item Optical Tube on Dovetail
			\item Finderscope
			\item Guide Scope
				\begin{todolist}
				\item Guidescope Camera
				\item RJ45 cable to mount.	
				\end{todolist}
			\item Optical Train
				\begin{todolist}
					\item Extender Rings (optional)
					\item Filter Wheel (optional)
					\item Camera
				\end{todolist}
			\item Connect Hand Controller
			\item Connect Power
			\item Balance
			\item Aim Telescope toward Polaris
			\item Engage Clutch (light)
			\end{todolist}
		
		\item Rough Polar Alignment
			\begin{todolist}
			
			\item  Power on.
			\item Disengage clutches and turn 90 degrees in DEC.
			\item Turn RA so that polar scope is oriented vertically.  
			\item Rough Polar Align through Polar Scope
			\item Return RA and DEC to Home position (toward Polaris).
			
			
			
	
		\end{todolist}	
	\end{itemize}
